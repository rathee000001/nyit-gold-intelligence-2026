def iso_utc(dt: Optional[datetime] = None) -> str:

def local_iso(dt: Optional[datetime] = None) -> str:
    dt = dt or utc_now()

        from zoneinfo import ZoneInfo

def get_git_commit(repo_root: Path) -> Optional[str]:

    except Exception:

def stable_hash_file(path: Path) -> str:
    if not path.exists() or not path.is_file():
        return "missing"
    h = hashlib.sha256()
    with path.open("rb") as f:
        for chunk in iter(lambda: f.read(1024 * 1024), b""):
            h.update(chunk)
    return "sha256:" + h.hexdigest()


def read_json(path: Path, default: Any = None) -> Any:
    if not path.exists():
        return default
    try:
        return json.loads(path.read_text(encoding="utf-8"))
    except Exception:
        return default


def write_json(path: Path, data: Any) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    path.write_text(json.dumps(data, indent=2, default=str), encoding="utf-8")


def write_csv_dicts(path: Path, rows: List[Dict[str, Any]]) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    keys: List[str] = []
    for row in rows:
        for key in row.keys():
            if key not in keys:
                keys.append(key)
    if not keys:
        keys = ["empty"]
    with path.open("w", newline="", encoding="utf-8") as f:
        writer = csv.DictWriter(f, fieldnames=keys, extrasaction="ignore")
        writer.writeheader()
        for row in rows:
            writer.writerow({key: row.get(key) for key in keys})

class Logger:
    def __init__(self, paths: Paths) -> None:
        self.paths = paths
        self.events: List[Dict[str, Any]] = []
        self.started = utc_now()

    def event(self, event: str, status: str = "ok", details: Optional[Dict[str, Any]] = None) -> None:
        row = {
            "timestamp_utc": iso_utc(),
            "event": event,
            "status": status,
            "details": details or {},
        }
        self.events.append(row)
        write_json(self.paths.timeline, self.events)

def stable_hash_file(path: Path) -> str:
    if not path.exists() or not path.is_file():
        return "missing"
    h = hashlib.sha256()
    with path.open("rb") as f:
        for chunk in iter(lambda: f.read(1024 * 1024), b""):
            h.update(chunk)
    return "sha256:" + h.hexdigest()

def write_csv_dicts(path: Path, rows: List[Dict[str, Any]]) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    keys: List[str] = []
    for row in rows:
        for key in row.keys():
            if key not in keys:
                keys.append(key)
    if not keys:
        keys = ["empty"]
    with path.open("w", newline="", encoding="utf-8") as f:
        writer = csv.DictWriter(f, fieldnames=keys, extrasaction="ignore")
        writer.writeheader()
        for row in rows:
            writer.writerow({key: row.get(key) for key in keys})

def build_model_ranking(repo_root: Path, horizons: List[int]) -> Dict[str, Any]:
    ranking_rows = []
    weights_by_horizon: Dict[str, Dict[str, float]] = {}

    for horizon in horizons:
        horizon_rows = []
        for model_key, source in MODEL_SOURCES.items():
            eval_path = repo_root / source["evaluation"]
            if not eval_path.exists():
                continue
            evaluation = load_evaluation(repo_root, source["evaluation"])
            test_mape = extract_test_mape(evaluation, horizon)
            val_mape = extract_validation_mape(evaluation, horizon)
            if val_mape is None and test_mape is None:
                continue
            score = val_mape if val_mape is not None else test_mape
            horizon_rows.append({
                "model_key": model_key,
                "horizon": horizon,
                "validation_mape_pct": val_mape,
                "test_mape_pct": test_mape,
                "weight_basis_mape_pct": score,
            })

        horizon_rows = sorted(horizon_rows, key=lambda x: (x["weight_basis_mape_pct"] is None, x["weight_basis_mape_pct"] or 1e9))
        denom = sum(1.0 / max(float(r["weight_basis_mape_pct"] or 999), 0.0001) for r in horizon_rows)
        horizon_weights = {}
        for idx, row in enumerate(horizon_rows, start=1):
            weight = (1.0 / max(float(row["weight_basis_mape_pct"] or 999), 0.0001)) / denom if denom else 0.0
            row["rank"] = idx
            row["omega_weight"] = weight
            horizon_weights[row["model_key"]] = weight
            ranking_rows.append(row)
        weights_by_horizon[str(horizon)] = horizon_weights

    return {
        "artifact_type": "omega_model_ranking",
        "schema_version": "1.0.0",
        "model_key": MODEL_KEY,
        "ranking_basis": "inverse_validation_mape_by_horizon_fallback_test_mape",
        "ranking": ranking_rows,
        "weights_by_horizon": weights_by_horizon,
        "professor_safe_note": "Weights are validation/test metric based fusion weights. They are not causal importance scores.",
        "generated_at_utc": iso_utc(),
    }

def build_omega_rollforward(repo_root: Path, weights_by_horizon: Dict[str, Dict[str, float]], smoke: bool) -> pd.DataFrame:
    frames = []
    for model_key, source in MODEL_SOURCES.items():
        df = load_rollforward(repo_root, model_key, source["rollforward"], smoke=smoke)
        if not df.empty:
            frames.append(df)
    if not frames:
        return pd.DataFrame()

    all_rows = pd.concat(frames, ignore_index=True)
    all_rows["date"] = pd.to_datetime(all_rows["date"], errors="coerce").dt.date.astype(str)
    all_rows["horizon"] = pd.to_numeric(all_rows["horizon"], errors="coerce").astype("Int64")
    all_rows["prediction"] = pd.to_numeric(all_rows["prediction"], errors="coerce")
    all_rows["actual_target"] = pd.to_numeric(all_rows["actual_target"], errors="coerce")
    all_rows["naive_prediction"] = pd.to_numeric(all_rows["naive_prediction"], errors="coerce")
    all_rows["gold_price"] = pd.to_numeric(all_rows["gold_price"], errors="coerce")

    def weighted_group(g: pd.DataFrame) -> pd.Series:
        horizon = str(int(g["horizon"].iloc[0]))
        weights = weights_by_horizon.get(horizon, {})
        valid = g.dropna(subset=["prediction"])
        if valid.empty:
            return pd.Series({})
        w = valid["source_model_key"].map(lambda k: weights.get(str(k), 0.0)).astype(float)
        if w.sum() <= 0:
            w = pd.Series([1.0 / len(valid)] * len(valid), index=valid.index)
        else:
            w = w / w.sum()
        pred = float((valid["prediction"] * w).sum())
        actual = valid["actual_target"].dropna().iloc[0] if valid["actual_target"].notna().any() else None
        naive = valid["naive_prediction"].dropna().iloc[0] if valid["naive_prediction"].notna().any() else None
        gold = valid["gold_price"].dropna().iloc[0] if valid["gold_price"].notna().any() else None
        return pd.Series({
            "gold_price": gold,
            "actual_target": actual,
            "prediction": pred,
            "naive_prediction": naive,
            "error": pred - actual if actual is not None and pd.notna(actual) else None,
            "model_count": int(len(valid)),
            "component_models": ",".join(valid["source_model_key"].astype(str).tolist()),
        })

    grouped = all_rows.groupby(["date", "split", "horizon"], dropna=True).apply(weighted_group).reset_index()
    return grouped.sort_values(["date", "horizon", "split"])


def build_omega_evaluation(omega: pd.DataFrame) -> Dict[str, Any]:
    out: Dict[str, Any] = {
        "artifact_type": "omega_evaluation_by_horizon",
        "schema_version": "1.0.0",
        "model_key": MODEL_KEY,
        "point_forecast": "weighted_fusion_prediction",
        "metrics_by_horizon": {},
        "generated_at_utc": iso_utc(),
    }
    if omega.empty:
        out["status"] = "no_omega_rows"
        return out

    for horizon, hdf in omega.groupby("horizon"):
        out["metrics_by_horizon"][str(int(horizon))] = {}
        for split, sdf in hdf.groupby("split"):
            y = pd.to_numeric(sdf["actual_target"], errors="coerce")
            pred = pd.to_numeric(sdf["prediction"], errors="coerce")
            mask = y.notna() & pred.notna()
            if not mask.any():
                continue
            err = pred[mask] - y[mask]
            ape = (err.abs() / y[mask].abs().replace(0, pd.NA)).dropna() * 100.0
            out["metrics_by_horizon"][str(int(horizon))][str(split)] = {
                "count": int(mask.sum()),
                "mae": float(err.abs().mean()),
                "rmse": float((err.pow(2).mean()) ** 0.5),
                "mape_pct": float(ape.mean()) if len(ape) else None,
                "bias_mean_error": float(err.mean()),
            }
    return out
